Docker N8N 운영 관리

안정적인 N8N 운영을 위한 Docker 핵심 관리법
데이터 영구 저장

Volume 마운트가 핵심

Docker 컨테이너를 재시작하면 내부 데이터가 사라집니다. Volume 마운트로 N8N 데이터(워크플로우·Credential)를 호스트에 영구 보존해야 합니다.

docker-compose.yml

한 파일로 전체 설정 관리

환경변수, 포트, 볼륨, 재시작 정책을 하나의 yml 파일로 관리합니다. docker compose up -d 한 줄로 실행 완료!

docker-compose.yml 
# docker-compose.yml — N8N 완전 설정
version: '3.8'

services:
  n8n:
    image: n8nio/n8n:latest
    container_name: n8n
    restart: unless-stopped   # 서버 재시작 시 자동 실행
    ports:
      - "5678:5678"
    environment:
      - N8N_BASIC_AUTH_ACTIVE=true     # 기본 인증 활성화
      - N8N_BASIC_AUTH_USER=admin
      - N8N_BASIC_AUTH_PASSWORD=비밀번호
      - GENERIC_TIMEZONE=Asia/Seoul    # 한국 시간대
      - N8N_HOST=0.0.0.0
      - N8N_PROTOCOL=http
      - WEBHOOK_URL=http://서버IP:5678/ # 외부 접근 URL
    volumes:
      - n8n_data:/home/node/.n8n       # 데이터 영구 저장 ✅

volumes:
  n8n_data:
Terminal — 핵심 운영 명령어

$ docker compose up -d          # 백그라운드 실행

$ docker compose restart n8n     # 재시작

# ─── 상태 확인 ───
$ docker compose ps              # 실행 중인 컨테이너
$ docker compose logs n8n        # 로그 확인
$ docker compose logs -f n8n     # 실시간 로그 스트리밍

$ docker compose pull n8n        # 최신 이미지 다운로드
$ docker compose up -d           # 새 버전으로 재실행

# ─── 데이터 백업 ───
$ docker run --rm -v n8n_data:/data -v $(pwd):/backup \

어느 노드 실패해도 관리자 텔레그램으로 오류 내용 + 실패 노드명 자동 발송

프로젝트 A — Code 노드 핵심 코드
// 날씨 + 환율 노드 데이터 병합
const weather = $("날씨 API").item.json;
const rates = $("환율 API").item.json.rates;

const now = new Date();
const kst = new Date(now.getTime() + 9 * 3600000);
const date = kst.toISOString().slice(0, 10);
const [y, m, d] = date.split('-');

const usd = Math.round(rates.KRW);
const eur = Math.round(rates.KRW / rates.EUR);
const jpy = Math.round(rates.KRW / rates.JPY * 100);
const temp = Math.round(weather.main.temp * 10) / 10;

// 텔레그램 메시지 (Markdown)
const tgMsg = `🌅 *일일 보고서* (${y}년 ${m}월 ${d}일)

🌤 서울 날씨: *${temp}°C* (${weather.weather[0].description})
💧 습도: ${weather.main.humidity}%

💱 *오늘의 환율*
🇺🇸 USD → *₩${usd.toLocaleString()}*
🇪🇺 EUR → *₩${eur.toLocaleString()}*
🇯🇵 JPY(100엔) → *₩${jpy.toLocaleString()}*

_구글 시트에 자동 기록 완료 ✅_`;

return [{
  json: { date, temp, weather: weather.weather[0].description,
    humidity: weather.main.humidity, usd, eur, jpy, tgMsg }
}];

// Webhook 수신 데이터
const order = $("Webhook Trigger").item.json.body;
const stock = items[0].json;  // 시트에서 읽은 현재 재고

const currentStock = Number(stock.재고수량);
const orderQty = Number(order.수량);
const newStock = currentStock - orderQty;

const now = new Date().toLocaleString('ko-KR');
const isLow = newStock < 10;

return [{
  json: {
    상품명: order.상품명,
    주문수량: orderQty,
    이전재고: currentStock,
    현재재고: newStock,
    업데이트시간: now,
    재고부족: isLow,
    경보메시지: isLow
      ? `⚠️ *재고 부족 경보!*\n\n📦 상품: ${order.상품명}\n현재 재고: *${newStock}개*\n즉시 발주가 필요합니다!`
      : null
  }
}];

return items;                                  ✅ Code 노드 하나로 처리

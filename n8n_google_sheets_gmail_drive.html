시트 이름 지정

Sheet Name vs Sheet ID

시트 이름은 탭 이름 그대로 입력 (예: "시트1", "환율데이터")

시트 읽기 → Code 노드로 통계 계산
// 시트에서 읽은 데이터로 통계 계산
const scores = items.map(i => i.json.점수);

const avg = scores.reduce((a, b) => a + b, 0) / scores.length;
const max = Math.max(...scores);
const min = Math.min(...scores);
const passing = scores.filter(s => s >= 80).length;

return [{
  json: {
    총인원: items.length,
    평균점수: Math.round(avg),
    최고점: max,
    최저점: min,
    합격자수: passing,
    합격률: Math.round(passing / items.length * 100) + "%"
  }
}];

Code 노드 — 날씨 + 환율 데이터 통합
// ─── 날씨 노드 데이터 참조 ───
const weather = $("날씨 API 호출").item.json;

// ─── 환율 노드 데이터 참조 ───
const rates = $("환율 API 호출").item.json.rates;

// 한국 시간
const now = new Date();
const kst = new Date(now.getTime() + 9 * 3600000);
const dateStr = kst.toISOString().split('T')[0];
const timeStr = kst.toISOString().slice(11, 16);

// 환율 계산
const USD_KRW = Math.round(rates.KRW);
const EUR_KRW = Math.round(rates.KRW / rates.EUR);
const JPY_KRW = Math.round(rates.KRW / rates.JPY * 100);

return [{
  json: {
    날짜:        dateStr,
    수집시간:    timeStr,
    도시:        weather.name,
    기온:        Math.round(weather.main.temp * 10) / 10,
    체감기온:    Math.round(weather.main.feels_like),
    날씨:        weather.weather[0].description,
    습도:        weather.main.humidity,
    USD_KRW:     USD_KRW,
    EUR_KRW:     EUR_KRW,
    JPY_KRW:     JPY_KRW
  }
}];

Gmail 노드

이메일 자동 발송

Google 계정으로 이메일을 보냅니다. 텍스트뿐 아니라 HTML 형식으로 표, 색상, 링크가 들어간 이메일도 전송 가능합니다.

HTML 이메일

보고서급 이메일 만들기

Code 노드에서 HTML 문자열을 생성 후, Gmail 노드의 Message Type: HTML로 설정하면 예쁜 보고서 이메일이 됩니다.

Gmail 노드 기본 설정
// ─── Gmail 노드 설정 ───
Resource:      Message
Operation:     Send

To:            {{ $json.이메일 }}      // 또는 고정: "user@example.com"
Subject:       {{ "[" + $now.format('MM/dd') + "] 일일 보고서" }}
Message Type:  HTML                 // HTML 형식 선택!
Message:       {{ $json.html_body }}  // Code 노드에서 생성한 HTML

// 첨부파일 추가 (선택)
Attachments:   Binary Data
Field Name:    data

Code 노드 — 시트 데이터 통계 계산
// Sheets Get Rows → Code 노드로 연결
const rates = items.map(i => Number(i.json.USD_KRW));

const avg = rates.reduce((a, b) => a + b, 0) / rates.length;
const max = Math.max(...rates);
const min = Math.min(...rates);
const latest = rates[rates.length - 1];
const change = latest - rates[0];  // 기간 대비 변동

return [{
  json: {
    기간:    `최근 ${rates.length}일`,
    평균:    Math.round(avg),
    최고:    max,
    최저:    min,
    최신:    latest,
    변동:    change > 0 ? `+${change}` : String(change),
    추세:    change > 0 ? "상승 📈" : change < 0 ? "하락 📉" : "보합 ➡️"
  }
}];
